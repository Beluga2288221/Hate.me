import { initializeApp } from "firebase/app"; import { getFirestore, doc, setDoc, getDoc, collection, query, where, getDocs, onSnapshot, addDoc, updateDoc, arrayUnion, orderBy } from "firebase/firestore"; // Данные теперь подтягиваются из настроек Vercel, а не лежат в открытом виде const firebaseConfig = { apiKey: import.meta.env.VITE_FIREBASE_API_KEY, authDomain: import.meta.env.VITE_FIREBASE_AUTH_DOMAIN, projectId: import.meta.env.VITE_FIREBASE_PROJECT_ID, storageBucket: import.meta.env.VITE_FIREBASE_STORAGE_BUCKET, appId: import.meta.env.VITE_FIREBASE_APP_ID }; const app = initializeApp(firebaseConfig); const db = getFirestore(app); // Основная логика входа и регистрации export async function doAuth(email, pass, mode) { try { const userRef = doc(db, "users", email.toLowerCase()); const userDoc = await getDoc(userRef); if (mode === 'login') { if (userDoc.exists() && userDoc.data().password === pass) { return { success: true, user: { ...userDoc.data(), email } }; } throw new Error("Неверный логин или пароль"); } else { if (userDoc.exists()) throw new Error("Пользователь уже существует"); // Создаем профиль с дефолтными данными const newUser = { name: "User", nick: email.split('@')[0], bio: "hate hate hate...", password: pass, myChats: [], av: "" }; await setDoc(userRef, newUser); return { success: true, user: { ...newUser, email } }; } } catch (e) { // Если в консоли Firebase не настроены Rules, будет ошибка прав console.error("Auth Error:", e.message); throw e; } } // Отправка сообщения в Firestore export async function sendMsg(meEmail, toEmail, text) { const chatID = [meEmail, toEmail].sort().join('_'); // Обновляем список чатов у обоих await updateDoc(doc(db, "users", meEmail), { myChats: arrayUnion({id: toEmail}) }); await updateDoc(doc(db, "users", toEmail), { myChats: arrayUnion({id: meEmail}) }); await addDoc(collection(db, `chats/${chatID}/messages`), { from: meEmail, text: text, time: Date.now() }); }
